// define o que vai ser rodado e seus parâmetros
$actions = (New-ScheduledTaskAction -Execute 'bomb.bat' -Argument Hidden)
// define o gatilho que desencadeará a ação
$trigger = New-ScheduledTaskTrigger -Daily -At '12:00 PM' -DaysInterval 2
// define o usuário e nível de privilégio
$principal = New-ScheduledTaskPrincipal -UserId 'Administrator' -RunLevel Highest
// cria a tarefa a partir dos dados anteriores 
$task = New-ScheduledTask -Action $actions -Principal $principal -Trigger $trigger 
// registra a tarefa no agendador de tarefas
Register-ScheduledTask 'bomb' -InputObject $task
